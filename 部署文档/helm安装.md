```
$curl -LO https://get.helm.sh/helm-v3.15.0-linux-amd64.tar.gz
$tar -zxvf helm-v3.15.0-linux-amd64.tar.gz
$mv linux-amd64/helm /usr/local/bin/helm  #将 linux-amd64/helm 文件移动到 /usr/local/bin/helm，并赋予它可执行权限

$helm version
```
